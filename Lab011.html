Node.js code for hello world application

Passive method 1

var http = require('http');                 // http mod

http.createServer(function (req, res) {     // create http server

	res.writeHead(200, {'content-type': 'text/plain'}); // content desc
	
	res.end("Hello, World!\n"); // write message
	
}).listen(8000);

console.log('Server running on 8000');

Read File method 2

var http = require('http'),
    fs = require('fs');

fs.readFile('./helloworld.html', function(err, html) { // reads content of Lab011.html which simply says "hello world"
    if (err) {
        throw err;
    }
    http.createServer(function (request, response) {
        response.writeHeader(200, {
            "Content-Type": "text/html"
        });
        response.write(html);
        response.end();
    }).listen(8000);
    console.log('listen: localhost:8000');
});
